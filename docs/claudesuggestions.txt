```python
import requests
from bs4 import BeautifulSoup
import ollama
import json
from typing import Optional

class SmartCrawler:
    def __init__(self, model: str = "qwen2.5:latest"):
        self.model = model
        self.extraction_prompt = """
You are a web data extraction expert. Extract faculty information from the HTML below.

HTML Content:
{html_content}

Extract the following as valid JSON matching this schema:
{schema}

Rules:
1. If a field is not found, use null or [] (not "N/A" or "Not found")
2. For education: Extract full degree strings (e.g., "Ph.D. Computer Science, Stanford, 2010")
3. For research_interests: Extract keywords/phrases, even if listed as "areas of expertise"
4. For bio_summary: Condense the bio to 2-3 sentences max
5. Return ONLY valid JSON, no markdown fences

JSON Output:
"""
    
    def extract_faculty_data(self, url: str) -> Optional[FacultyProfile]:
        """
        Fetch page and extract structured data using Qwen
        """
        # Fetch HTML
        response = requests.get(url, timeout=10)
        soup = BeautifulSoup(response.content, 'html.parser')
        
        # Clean HTML (remove scripts, styles, nav)
        for tag in soup(['script', 'style', 'nav', 'footer', 'header']):
            tag.decompose()
        
        # Limit to main content (heuristic: find largest text block)
        main_content = soup.find('main') or soup.find('article') or soup.body
        html_text = main_content.get_text(separator='\n', strip=True)[:8000]  # Qwen context limit
        
        # Generate extraction prompt
        schema_json = FacultyProfile.schema_json(indent=2)
```

```python
# db/vector_store.py
import lancedb
from sentence_transformers import SentenceTransformer
import pyarrow as pa

class VectorStore:
    def __init__(self, db_path: str = "./lancedb"):
        self.db = lancedb.connect(db_path)
        self.embedder = SentenceTransformer('nomic-ai/nomic-embed-text-v1.5', trust_remote_code=True)
        
        # Define schema (Lance uses Arrow schema)
        self.schema = pa.schema([
            pa.field("id", pa.string()),
            pa.field("title", pa.string()),
            pa.field("authors", pa.list_(pa.string())),
            pa.field("abstract", pa.string()),
            pa.field("full_text", pa.string()),  # Abstract + Intro + Conclusion
            pa.field("embedding", pa.list_(pa.float32(), 768)),  # Nomic dim
            pa.field("metadata", pa.struct([
                ('year', pa.int32()),
                ('venue', pa.string()),
                ('citations', pa.int32())
            ]))
        ])
    
    def create_table(self, name: str = "papers"):
        """Initialize papers table"""
        if name not in self.db.table_names():
            self.table = self.db.create_table(name, schema=self.schema)
        else:
            self.table = self.db.open_table(name)
    
    def add_papers(self, papers: List[dict]):
        """
        Batch insert papers with embeddings
        """
        # Prepare full text for embedding
        full_texts = [
            f"{p['abstract']} {p.get('intro', '')} {p.get('conclusion', '')}"[:8192]
            for p in papers
        ]
        
        # Generate embeddings in batch (much faster)
        embeddings = self.embedder.encode(
            full_texts,
            batch_size=32,
            show_progress_bar=True
        )
        
        # Format for Lance
        data = [
            {
                "id": p['id'],
                "title": p['title'],
                "authors": p['authors'],
                "abstract": p['abstract'],
                "full_text": full_texts[i],
                "embedding": embeddings[i].tolist(),
                "metadata": {
                    'year': p.get('year', 0),
                    'venue': p.get('venue', ''),
                    'citations': p.get('citations', 0)
                }
            }
            for i, p in enumerate(papers)
        ]
        
        self.table.add(data)
    
    def search(self, query: str, top_k: int = 10, filters: dict = None):
        """
        Semantic search with optional metadata filtering
        """
        query_embedding = self.embedder.encode([query])[0]
        
        # LanceDB syntax for filtered search
        results = self.table.search(query_embedding.tolist()) \
            .limit(top_k)
        
        if filters:
            # Example: filters = {'year': ('>=', 2020)}
            for field, (op, value) in filters.items():
                results = results.where(f"metadata.{field} {op} {value}")
        
        return results.to_pandas()
```

```python
# retrieval/query_rewriter.py
import ollama
from typing import List

class QueryRewriter:
    def __init__(self, model: str = "qwen2.5:latest"):
        self.model = model
        self.expansion_prompt = """
You are a research query expansion expert. Given a user's search query, generate 3 semantically similar variations that would help find relevant academic papers.

Rules:
1. Include synonyms and related technical terms
2. Expand acronyms (e.g., "ML" → "machine learning")
3. Add domain context (e.g., "fairness" → "algorithmic fairness in AI")
4. Keep queries concise (5-10 words each)
5. Return as JSON array of strings

Original Query: {query}

Expanded Queries (JSON array):
"""
    
    def expand_query(self, query: str) -> List[str]:
        """
        Generate query variations for better recall
        """
        prompt = self.expansion_prompt.format(query=query)
        
        response = ollama.generate(
            model=self.model,
            prompt=prompt,
            format='json',
            options={'temperature': 0.7}  # Higher temp for diversity
        )
        
        try:
            expanded = json.loads(response['response'])
            return [query] + expanded  # Original + variations
        except:
            return [query]  # Fallback to original only
    
    def rewrite_with_context(self, query: str, conversation_history: List[str]) -> str:
        """
        Rewrite query using conversation context (for follow-up questions)
        """
        context_prompt = f"""
Previous questions: {conversation_history}
Current question: {query}

Rewrite the current question as a standalone query that includes necessary context from the conversation.

Standalone Query:
"""
        response = ollama.generate(
            model=self.model,
            prompt=context_prompt,
            options={'temperature': 0.3}
        )
        
        return response['response'].strip()
```

**B. Integrate with Search Pipeline**
```python
# retrieval/search_pipeline.py
from db.vector_store import VectorStore
from retrieval.query_rewriter import QueryRewriter
import pandas as pd

class SearchPipeline:
    def __init__(self):
        self.vector_store = VectorStore()
        self.rewriter = QueryRewriter()
    
    def search(self, query: str, top_k: int = 20, use_expansion: bool = True):
        """
        Multi-query search with result fusion
        """
        if use_expansion:
            queries = self.rewriter.expand_query(query)
            print(f"Expanded to {len(queries)} queries: {queries}")
        else:
            queries = [query]
        
        # Search with each query
        all_results = []
        for q in queries:
            results = self.vector_store.search(q, top_k=top_k)
            results['query'] = q
            all_results.append(results)
        
        # Reciprocal Rank Fusion (RRF)
        combined = self._reciprocal_rank_fusion(all_results)
        return combined.head(top_k)
    
    def _reciprocal_rank_fusion(self, results_list: List[pd.DataFrame], k: int = 60):
        """
        Combine multiple result sets using RRF algorithm
        """
        scores = {}
        for results in results_list:
            for rank, row in enumerate(results.itertuples(), start=1):
                doc_id = row.id
                if doc_id not in scores:
                    scores[doc_id] = {'score': 0, 'data': row}
                scores[doc_id]['score'] += 1 / (k + rank)
        
        # Sort by fused score
        sorted_results = sorted(scores.items(), key=lambda x: x[1]['score'], reverse=True)
        
        # Convert back to DataFrame
        return pd.DataFrame([item[1]['data'] for item in sorted_results])
```

```python
    """Calculate recall@20"""
    retrieved_titles = results['title'].tolist()

    return hits / len(expected)
```

```python
# graph/relation_extractor.py
import ollama
import networkx as nx
from typing import List, Tuple

class RelationExtractor:
    def __init__(self, model: str = "qwen2.5:latest"):
        self.model = model
        self.extraction_prompt = """
You are a knowledge graph extraction expert. Extract entities and their relationships from the research paper text below.

Text:
{text}

Extract:
1. **Entities**: Key concepts, methods, datasets, metrics (e.g., "BERT", "ImageNet", "F1-score")
2. **Relations**: How entities connect (e.g., "BERT" --[trained_on]--> "Wikipedia")

Relation types: uses, trained_on, improves, evaluates, proposes, extends, compares_to

Return as JSON:
{{
  "entities": [
    {{"id": "entity_1", "type": "model", "name": "BERT"}},
    {{"id": "entity_2", "type": "dataset", "name": "Wikipedia"}}
  ],
  "relations": [
    {{"source": "entity_1", "target": "entity_2", "type": "trained_on"}}
  ]
}}

JSON Output:
"""
    
    def extract_from_paper(self, paper_text: str) -> dict:
        """
        Extract knowledge graph from paper text
        """
        # Chunk text if too long (Qwen limit: ~32k tokens)
        chunks = self._chunk_text(paper_text, max_length=6000)
        
        all_entities = []
        all_relations = []
        
        for chunk in chunks:
            prompt = self.extraction_prompt.format(text=chunk)
            response = ollama.generate(
                model=self.model,
                prompt=prompt,
                format='json',
                options={'temperature': 0.2}
            )
            
            try:
                data = json.loads(response['response'])
                all_entities.extend(data['entities'])
                all_relations.extend(data['relations'])
            except:
                continue
        
        # Deduplicate entities by name
        unique_entities = {e['name']: e for e in all_entities}.values()
        
        return {
            'entities': list(unique_entities),
            'relations': all_relations
        }
    
    def _chunk_text(self, text: str, max_length: int) -> List[str]:
        """Split text into chunks at sentence boundaries"""
        sentences = text.split('. ')
        chunks = []
        current_chunk = []
        current_length = 0
        
        for sent in sentences:
            if current_length + len(sent) > max_length:
                chunks.append('. '.join(current_chunk))
                current_chunk = [sent]
                current_length = len(sent)
            else:
                current_chunk.append(sent)
                current_length += len(sent)
        
        if current_chunk:
            chunks.append('. '.join(current_chunk))
        
        return chunks
```

**B. Graph Builder**
```python
# graph/knowledge_graph.py
import networkx as nx
import pickle
from typing import List, Dict

class KnowledgeGraph:
    def __init__(self):
        self.graph = nx.MultiDiGraph()  # Allows multiple edges between nodes
    
    def add_paper(self, paper_id: str, extraction_result: dict):
        """
        Add paper's entities and relations to graph
        """
        # Add paper node
        self.graph.add_node(
            paper_id,
            type='paper',
            **extraction_result.get('metadata', {})
        )
        
        # Add entities
        for entity in extraction_result['entities']:
            entity_id = entity['id']
            self.graph.add_node(
                entity_id,
                name=entity['name'],
                type=entity['type']
            )
            
            # Connect entity to paper
            self.graph.add_edge(
                paper_id,
                entity_id,
                relation='mentions'
            )
        
        # Add relations between entities
        for rel in extraction_result['relations']:
            self.graph.add_edge(
                rel['source'],
                rel['target'],
                relation=rel['type'],
                paper_source=paper_id  # Track where relation came from
            )
    
    def find_related_concepts(self, concept: str, max_depth: int = 2) -> List[str]:
        """
        Find concepts related to the query concept via graph traversal
        """
        # Find all nodes matching concept name (fuzzy match)
        matching_nodes = [
            node for node, data in self.graph.nodes(data=True)
            if 'name' in data and concept.lower() in data['name'].lower()
        ]
        
        if not matching_nodes:
            return []
        
        # BFS to find related concepts
        related = set()
        for node in matching_nodes:
            # Get neighbors within max_depth hops
            for target in nx.single_source_shortest_path_length(
                self.graph, node, cutoff=max_depth
            ):
                if self.graph.nodes[target].get('type') in ['model', 'method', 'concept']:
                    related.add(self.graph.nodes[target]['name'])
        
        return list(related)
    
    def find_connection_path(self, concept_a: str, concept_b: str) -> List[Tuple]:
        """
        Find shortest path explaining how two concepts are connected
        """
        # Find matching nodes
        nodes_a = [n for n, d in self.graph.nodes(data=True) 
                   if d.get('name', '').lower() == concept_a.lower()]
        nodes_b = [n for n, d in self.graph.nodes(data=True) 
                   if d.get('name', '').lower() == concept_b.lower()]
        
        if not nodes_a or not nodes_b:
            return []
        
        try:
            path = nx.shortest_path(self.graph, nodes_a[0], nodes_b[0])
            
            # Format as (source, relation, target) triples
            path_triples = []
            for i in range(len(path) - 1):
                edge_data = self.graph.get_edge_data(path[i], path[i+1])
                relation = list(edge_data.values())[0]['relation']
                path_triples.append((
                    self.graph.nodes[path[i]]['name'],
                    relation,
                    self.graph.nodes[path[i+1]]['name']
                ))
            
            return path_triples
        except nx.NetworkXNoPath:
            return []
    
    def save(self, filepath: str = "./knowledge_graph.pkl"):
        """Persist graph to disk"""
        with open(filepath, 'wb') as f:
            pickle.dump(self.graph, f)
    
    def load(self, filepath: str = "./knowledge_graph.pkl"):
        """Load graph from disk"""
        with open(filepath, 'rb') as f:
            self.graph = pickle.load(f)
```

```python
    def export_to_cytoscape_json(self, subgraph: nx.Graph) -> dict:
        """
        Export for Cytoscape.js (web embedding)
        """
        elements = []
        
        # Add nodes
        for node, data in subgraph.nodes(data=True):
            elements.append({
                'data': {
                    'id': node,
                    'label': data.get('name', node),
                    'type': data.get('type', 'unknown')
                }
            })
        
        # Add edges
        for source, target, data in subgraph.edges(data=True):
            elements.append({
                'data': {
                    'source': source,
                    'target': target,
                    'label': data.get('relation', 'related')
                }
            })
        
        return {'elements': elements}
```

### 2.3 A/B Test: Vector-Only vs GraphRAG Retrieval
**Duration**: 3-4 days

#### Implementation Steps:

**A. GraphRAG Retrieval Pipeline**
```python
# retrieval/graph_rag.py
from graph.knowledge_graph import KnowledgeGraph
from db.vector_store import VectorStore
from retrieval.query_rewriter import QueryRewriter
import pandas as pd

class GraphRAG:
    def __init__(self):
        self.kg = KnowledgeGraph()
        self.kg.load()
        self.vs = VectorStore()
        self.rewriter = QueryRewriter()
    
    def search_with_graph(self, query: str, top_k: int = 10):
        """
        Two-stage retrieval: Graph expansion → Vector search
        """
        # Stage 1: Extract concepts from query
        concept_extraction_prompt = f"""
Extract key technical concepts from this query: "{query}"
Return as JSON array of concept names.
Example: ["neural networks", "image classification"]
"""
        response = ollama.generate(
            model="qwen2.5:latest",
            prompt=concept_extraction_prompt,
            format='json'
        )
        
        query_concepts = json.loads(response['response'])
        
        # Stage 2: Graph expansion
        expanded_concepts = set(query_concepts)
        for concept in query_concepts:
            related = self.kg.find_related_concepts(concept, max_depth=1)
            expanded_concepts.update(related[:5])  # Top 5 related per concept
        
        # Stage 3: Build enriched query
        enriched_query = f"{query} {' '.join(expanded_concepts)}"
        
        # Stage 4: Vector search
        results = self.vs.search(enriched_query, top_k=top_k * 2)  # Retrieve 2x, will rerank
        
        # Stage 5: Rerank by graph connectivity
        results['graph_score'] = results.apply(
            lambda row: self._calculate_graph_score(row['id'], query_concepts),
            axis=1
        )
        
        # Combine vector similarity + graph score
        results['final_score'] = 0.7 * results['_distance'] + 0.3 * results['graph_score']
        results = results.sort_values('final_score', ascending=False)
        
        return results.head(top_k)
    
    def _calculate_graph_score(self, paper_id: str, query_concepts: List[str]) -> float:
        """
        Score paper based on how many query concepts it connects
        """
        if paper_id not in self.kg.graph:
            return 0.0
        
        # Get paper's entities
        paper_entities = [
            n for n in self.kg.graph.neighbors(paper_id)
            if self.kg.graph.nodes[n].get('type') != 'paper'
        ]
        
        # Count matches with query concepts
        matches = 0
        for entity in paper_entities:
            entity_name = self.kg.graph.nodes[entity].get('name', '').lower()
            for concept in query_concepts:
                if concept.lower() in entity_name:
                    matches += 1
        
        return matches / len(query_concepts) if query_concepts else 0.0
```

```python
def calculate_metrics(results: pd.DataFrame, relevant: List[str]) -> dict:
    """Calculate precision, recall, NDCG@10"""
    retrieved = results['id'].tolist()[:10]
    
    # Precision & Recall
    tp = len(set(retrieved) & set(relevant))
    precision = tp / len(retrieved) if retrieved else 0
    recall = tp / len(relevant) if relevant else 0
    
    # NDCG@10
    dcg = sum([
        1 / np.log2(i + 2) if retrieved[i] in relevant else 0
        for i in range(len(retrieved))
    ])
    idcg = sum([1 / np.log2(i + 2) for i in range(min(len(relevant), 10))])
    ndcg = dcg / idcg if idcg > 0 else 0
    
    return {'precision': precision, 'recall': recall, 'ndcg': ndcg}
```

```python
# retrieval/reranker.py
from sentence_transformers import CrossEncoder
import pandas as pd

class Reranker:
    def __init__(self, model: str = 'cross-encoder/ms-marco-MiniLM-L-12-v2'):
        """
        Cross-encoder for reranking top-k results
        """
        self.model = CrossEncoder(model)
    
    def rerank(self, query: str, results: pd.DataFrame, top_k: int = 10) -> pd.DataFrame:
        """
        Rerank results using query-document cross-attention
        """
        # Prepare query-document pairs
        pairs = [
            [query, f"{row['title']} {row['abstract']}"]
            for _, row in results.iterrows()
        ]
        
        # Score pairs
        scores = self.model.predict(pairs)
        
        # Add reranking scores
        results['rerank_score'] = scores
        results = results.sort_values('rerank_score', ascending=False)
        
        return results.head(top_k)
```

```python
import kuzu

class KuzuKnowledgeGraph:
    def __init__(self, db_path: str = "./kuzudb"):
        self.db = kuzu.Database(db_path)
        self.conn = kuzu.Connection(self.db)
        
        # Create schema
        self.conn.execute("""
            CREATE NODE TABLE IF NOT EXISTS Entity(
                id STRING,
                name STRING,
                type STRING,
                PRIMARY KEY(id)
            )
        """)
        
        self.conn.execute("""
            CREATE REL TABLE IF NOT EXISTS Relation(
                FROM Entity TO Entity,
                relation_type STRING,
                paper_source STRING
            )
        """)
    
    def add_entities(self, entities: List[dict]):
        """Batch insert entities"""
        for entity in entities:
            self.conn.execute(
                "CREATE (:Entity {id: $id, name: $name, type: $type})",
                parameters=entity
            )
    
    def find_related_concepts(self, concept: str, max_depth: int = 2):
        """Cypher query for traversal"""
        query = f"""
            MATCH (start:Entity {{name: '{concept}'}})-[*1..{max_depth}]-(related:Entity)
            WHERE related.type IN ['model', 'method', 'concept']
            RETURN DISTINCT related.name AS name
            LIMIT 20
        """
        results = self.conn.execute(query)
        return [row[0] for row in results.get_as_pl()]
```

```python
# ingestion/batch_processor.py
from concurrent.futures import ProcessPoolExecutor, as_completed
from ingestion.distill_paper import distill_paper
import multiprocessing as mp

class BatchProcessor:
    def __init__(self, num_workers: int = None):
        self.num_workers = num_workers or mp.cpu_count()
    
    def process_papers_parallel(self, pdf_paths: List[str]):
        """
        Process multiple PDFs in parallel
        """
        with ProcessPoolExecutor(max_workers=self.num_workers) as executor:
            # Submit all tasks
            futures = {
                executor.submit(distill_paper, pdf): pdf
                for pdf in pdf_paths
            }
            
            # Collect results as they complete
            results = []
            for future in as_completed(futures):
                pdf_path = futures[future]
                try:
                    result = future.result(timeout=120)  # 2 min timeout
                    results.append(result)
                    print(f"✓ Processed: {pdf_path}")
                except Exception as e:
                    print(f"✗ Failed: {pdf_path} - {e}")
            
            return results
```
